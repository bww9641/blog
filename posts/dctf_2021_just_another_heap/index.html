def relive(idx):
  p.recv()

  p.recv()
  p.sendline(str(4))
  p.recv()

  p.send(cont)

  p.recv()
  p.sendline(str(3))
  p.recv()
  p.sendline(str(idx))

  p.recv()
  p.sendline(str(5))
